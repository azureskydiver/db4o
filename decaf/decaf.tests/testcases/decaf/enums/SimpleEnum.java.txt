package decaf.enums;

enum Animal {
	Cow,
	Cat,
	Dog
}

public class SimpleEnum {
	
	public String simpleSwitch(Animal animal) {
		switch(theAnimal()) {
			case Cow:
			case Cat:
			case Dog:
				return "moo or meeow or wauwau";
		}

		switch(animal) {
			case Cow:
			case Cat:
				return "moo or meeow";
			case Dog:
				return "wauwau";
			default:
				throw new IllegalArgumentException();
		}
	}

	private Animal theAnimal() {
		return Animal.Dog;
	}
}
