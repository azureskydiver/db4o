package decaf.enums;

enum Animal {
	Cow,
	Cat,
	Dog
}

public class SimpleEnum {
	
	public String simpleSwitch(Animal animal) {
		switch(animal) {
			case Cow:
			case Cat:
				return "moo or meeow";
			case Dog:
				return "wauwau";
			default:
				throw new IllegalArgumentException();
		}
	}

}
