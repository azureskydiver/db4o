package decaf.enums;

enum Hero {
	SuperMan(1938) {
		public String movieName() {
			return "Superman Returns"; 
		}
	}
	
	Batman(1939) {
		public String movieName() {
			return "The Incredibles"; 
		}
	}
	
	BatGirl(1961) {
		public String movieName() {
			return "The Incredibles"; 
		}
	}
	
	BobParr(2004) {
		public String movieName() {
			return "The Incredibles"; 
		}
	}

	Hero(int year) {
		_year = year; 
	}
	
	public int year() {
		return _year; 
	}
	
	public abstract String movieName();
	
	private final int _year;	
}
