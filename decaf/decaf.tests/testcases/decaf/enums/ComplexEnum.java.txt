package decaf.enums;

enum Hero {
	SuperMan(1938),
	Batman(1939),
	BatGirl(1961),
	BobParr(2004);

	Hero(int year) {
		_year = year; 
	}
	
	public int year() {
		return _year; 
	}
	
	private final int _year;	
}

class TestEnum {
	public void testValueOf() {
		Hero h = Enum.valueOf(Hero.class, "BatGirl");
		Animal a = Enum.valueOf(Animal.class, "Dog");
	}
	
	public void testSwitch(Hero h) {
		switch(h) {
			case BobParr:
			case SuperMan: 
				System.out.println("SuperMan or Mr. Incredible"); 
				break;
				  		
			case BatGirl: 
				System.out.println("BatGirl"); 
				break;
			
			default: 
				System.out.println(h); 
				break;  		
		}
	}
}
