package decaf.enums;

final class Animal extends Enum4 {
	public static class Cow extends Animal {
		public static final int ORDINAL = 0;

		private Cow() {
			super("Cow", Cow.ORDINAL);
		}
	}

	public static class Cat extends Animal {
		public static final int ORDINAL = 1;

		private Cat() {
			super("Cat", Cat.ORDINAL);
		}
	}

	public static class Dog extends Animal {
		public static final int ORDINAL = 2;

		private Dog() {
			super("Dog", Dog.ORDINAL);
		}
	}

	public static final Animal Cow = new Cow();
	public static final Animal Cat = new Cat();
	public static final Animal Dog = new Dog();

	private Animal(String name, int ordinal) {
		super(name, ordinal);
	}

	public static Animal[] values() {
		return new Animal[] { Animal.Cow, Animal.Cat, Animal.Dog };
	}
}
