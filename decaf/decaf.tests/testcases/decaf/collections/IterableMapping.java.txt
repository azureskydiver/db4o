package decaf.collections;

class MyIterable implements Iterable<Integer> {
	Iterator<Integer> iterator() {
		return new List<Integer>().iterator();
	}
}

class SimpleIterable {
	public void test() {
		justIterate(new List<int>());		
		justIterate(new List<String>());
		justIterate(new MyIterable());
	}
		
	private void justIterate(Iterable<int> iterable) {
		for (int v : iterable) {
			System.out.println(v);
		}
	}
	
	private void justIterate2<T>(Iterable<T> iterable) {
		for (T v : iterable) {
			System.out.println("First Pass: " + v);
		}
	
		for (T v2 : iterable) {
			System.out.println("Second Pass: " + v2);
		}
	}		
}
