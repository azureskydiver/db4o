package decaf;

using java.lang.*;

class TestDriver {
	public <T> void GenericForEachArray(T []items) {
		for(T item : items) {
			System.out.println(item);
		}
	}
	
	public <T> void GenericForEachIterable(Iterable<T> items) {
		for(T item : items) {
			System.out.println(item);
		}
	}
}
