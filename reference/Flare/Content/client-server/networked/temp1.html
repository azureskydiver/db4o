Simple db4o Server

Let's implement a simple standalone db4o server with a special client that can tell the server to shut itself down gracefully on demand.

First, both the client and the server need some shared configuration information. We will provide this using an interface:

002package com.db4odoc.clientserver;
003
004import com.db4o.ObjectContainer;
005import com.db4o.ObjectServer;
006import com.db4o.cs.Db4oClientServer;
007import com.db4o.cs.config.ServerConfiguration;
008import com.db4o.messaging.MessageContext;
009import com.db4o.messaging.MessageRecipient;
010
011/** *//**
012 * starts a db4o server with the settings from {@link ServerSetup}. <br>
013 * <br>
014 * This is a typical setup for a long running server. <br>
015 * <br>
016 * The Server may be stopped from a remote location by running StopServer. The
017 * StartServer instance is used as a MessageRecipient and reacts to receiving an
018 * instance of a StopServer object. <br>
019 * <br>
020 * Note that all user classes need to be present on the server side and that all
021 * possible Db4oEmbedded.configure() calls to alter the db4o configuration need
022 * to be executed on the client and on the server.
023 */
024public class StartServer implements ServerSetup, MessageRecipient ...{
025
026  /** *//**
027   * setting the value to true denotes that the server should be closed
028   */
029  private boolean stop = false;
030
031  /** *//**
032   * starts a db4o server using the configuration from
033   * {@link ServerSetup}.
034   */
035  public static void main(String[] arguments) ...{
036    new StartServer().runServer();
037  }
038
039  // end main
040
041  /** *//**
042   * opens the ObjectServer, and waits forever until close() is called or a
043   * StopServer message is being received.
044   */
045  public void runServer() ...{
046    synchronized (this) ...{
047      // Using the messaging functionality to redirect all
048      // messages to this.processMessage
049      ServerConfiguration config = Db4oClientServer.newServerConfiguration();
050      config.networking().messageRecipient(this);
051
052      ObjectServer db4oServer = Db4oClientServer.openServer(
053          config, FILE, PORT);
054      db4oServer.grantAccess(USER, PASS);
055
056      
057      // to identify the thread in a debugger
058      Thread.currentThread().setName(this.getClass().getName());
059
060      // We only need low priority since the db4o server has
061      // it's own thread.
062      Thread.currentThread().setPriority(Thread.MIN_PRIORITY);
063      try ...{
064        if (!stop) ...{
065          // wait forever for notify() from close()
066          this.wait(Long.MAX_VALUE);
067        }
068      } catch (Exception e) ...{
069        e.printStackTrace();
070      }
071      db4oServer.close();
072    }
073  }
074
075  // end runServer
076
077  /** *//**
078   * messaging callback
079   * 
080   * @see com.db4o.messaging.MessageRecipient#processMessage(ObjectContainer,
081   *      Object)
082   */
083  public void processMessage(MessageContext context, Object message) ...{
084    if (message instanceof StopServer) ...{
085      close();
086    }
087  }
088
089  // end processMessage
090
091  /** *//**
092   * closes this server.
093   */
094  public void close() ...{
095    synchronized (this) ...{
096      stop = true;
097      this.notify();
098    }
099  }
100  // end close
101}

StopServer.java
01/**//* Copyright (C) 2007 Versant Inc. http://www.db4o.com */
02package com.db4odoc.clientserver;
03
04import com.db4o.*;
05import com.db4o.cs.Db4oClientServer;
06import com.db4o.messaging.*;
07
08/** *//**
09 * stops the db4o Server started with {@link StartServer}. <br>
10 * <br>
11 * This is done by opening a client connection to the server and by sending a
12 * StopServer object as a message. {@link StartServer} will react in it's
13 * processMessage method.
14 */
15public class StopServer implements ServerSetup ...{
16
17  /** *//**
18   * stops a db4o Server started with StartServer.
19   * 
20   * @throws Exception
21   */
22  public static void main(String[] args) ...{
23    ObjectContainer objectContainer = null;
24    try ...{
25
26      // connect to the server
27      objectContainer = Db4oClientServer.openClient(Db4oClientServer
28          .newClientConfiguration(), HOST, PORT, USER, PASS);
29
30    } catch (Exception e) ...{
31      e.printStackTrace();
32    }
33
34    if (objectContainer != null) ...{
35
36      // get the messageSender for the ObjectContainer
37      MessageSender messageSender = objectContainer.ext().configure()
38          .clientServer().getMessageSender();
39
40      // send an instance of a StopServer object
41      messageSender.send(new StopServer());
42
43      // close the ObjectContainer
44      objectContainer.close();
45    }
46  }
47  // end main
48}
