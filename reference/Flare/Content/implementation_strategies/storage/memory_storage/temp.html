Storing MemoryBin Data

In the previous example we've learned how to use a MemoryStorage to create a database and run queries in it. However, in most of the cases you will still need a backup of the data in memory earlier or later. To get it, you will need to close the object container and store the contents of the MemoryBin to a normal file.

MemoryStorageExample.java: storeMemoryBin
01public static void storeMemoryBin() throws IOException ...{
02    MemoryStorage origStorage = new MemoryStorage();
03    EmbeddedConfiguration origConfig = Db4oEmbedded.newConfiguration();
04    origConfig.file().storage(origStorage);
05    ObjectContainer origDb = Db4oEmbedded.openFile(origConfig, DB4O_URI);
06    origDb.store(new Book("Alice in Wonderland"));
07    origDb.close();
08
09    MemoryBin origBin = origStorage.bin(DB4O_URI);
10    byte[] data = origBin.data();
11    OutputStream out = new FileOutputStream("memory.db4o");
12    out.write(data, 0, data.length);
13    out.close();
14
15    ObjectContainer newDb = Db4oEmbedded.openFile(Db4oEmbedded
16        .newConfiguration(), "memory.db4o");
17    Book book = (Book) newDb
18        .queryByExample(new Book("Alice in Wonderland")).get(0);
19    System.out.println(book);
20    newDb.close();
21  }

01public static void StoreMemoryBin()
02        ...{
03            MemoryStorage origStorage = new MemoryStorage();
04            IEmbeddedConfiguration origConfig = Db4oEmbedded.NewConfiguration();
05            origConfig.File.Storage = origStorage;
06            IObjectContainer origDb = Db4oEmbedded.OpenFile(origConfig, Db4oUri);
07            origDb.Store(new Book("Alice in Wonderland"));
08            origDb.Close();
09
10            MemoryBin origBin = origStorage.Bin(Db4oUri);
11            byte[] data = origBin.Data();
12            FileStream fs1 = new FileStream("memory.db4o", FileMode.OpenOrCreate,
13    FileAccess.Write);
14            BinaryWriter bw = new BinaryWriter(fs1);
15            bw.Write(data);
16            bw.Close();
17            fs1.Close();
18
19
20            IObjectContainer newDb = Db4oEmbedded.OpenFile(Db4oEmbedded
21                    .NewConfiguration(), "memory.db4o");
22            Book book = (Book)newDb
23                    .QueryByExample(new Book("Alice in Wonderland"))[0];
24            System.Console.WriteLine(book);
25            newDb.Close();
26        }
