Complexity Of Objects

More complex objects are usually more difficult not only to store, but also to query and instantiate. The following test demonstrates how query performance depends on class structure, complexity and depth:

QueryPerformanceBenchmark.cs: RunDifferentObjectsTest
001private void RunDifferentObjectsTest()
002        ...{
003
004            Init();
005            System.Console.WriteLine("Storing " + _count + " objects with " + _depth
006                    + " levels of embedded objects:");
007
008            Clean();
009            System.Console.WriteLine();
010            System.Console.WriteLine(" - primitive object with int field");
011            Open(Configure());

015            Open(Configure());

017            IQuery query = objectContainer.Query();
018            query.Constrain(typeof(SimplestItem));
019            query.Descend("_id").Constrain(1);
020            IList result = query.Execute();
021            SimplestItem simplestItem = (SimplestItem)result[0];
022            StopTimer("Querying SimplestItem: " + simplestItem._id);
023            Close();
024
025            Open(Configure());
026            System.Console.WriteLine();
027            System.Console.WriteLine(" - object with string field");
028            Store();

030            Close();

032            StartTimer();
033            query = objectContainer.Query();
034            query.Constrain(typeof(Item));
035            query.Descend("_name").Constrain("level1/2");
036            result = query.Execute();
037            Item item = (Item)result[0];
038            StopTimer("Querying object with string field: " + item._name);
039            Close();
040
041            Clean();
042            Open(Configure());
043            System.Console.WriteLine();
044            System.Console.WriteLine(" - object with StringBuilder field");
045            StoreWithStringBuffer();
046            objectContainer.Ext().Purge();
047            Close();
048            Open(Configure());
049            StartTimer();
050            query = objectContainer.Query();
051            query.Constrain(typeof(ItemWithStringBuilder));
052            query.Descend("_name").Constrain(new StringBuilder("level1/2"));
053            result = query.Execute();
054            ItemWithStringBuilder itemWithSB = (ItemWithStringBuilder)result[0];
055            StopTimer("Querying object with StringBuilder field: "
056                    + itemWithSB._name);
057            Close();
058
059            Clean();
060            Open(Configure());
061            System.Console.WriteLine();
062            System.Console.WriteLine(" - object with int array field");
063            StoreWithArray();
064            objectContainer.Ext().Purge();
065            Close();
066            Open(Configure());
067            StartTimer();
068            query = objectContainer.Query();
069            query.Constrain(typeof(ItemWithArray));
070            IQuery idQuery = query.Descend("_id");
071            idQuery.Constrain(1);
072            idQuery.Constrain(2);
073            idQuery.Constrain(3);
074            idQuery.Constrain(4);
075            result = query.Execute();
076
077            ItemWithArray itemWithArray = (ItemWithArray)result[0];
078            StopTimer("Querying object with Array field: [" + itemWithArray._id[0]
079                    + ", " + +itemWithArray._id[1] + ", " + +itemWithArray._id[2]
080                    + ", " + +itemWithArray._id[0] + "]");
081            Close();
082
083            Clean();
084            Open(Configure());
085            System.Console.WriteLine();
086            System.Console.WriteLine(" - object with ArrayList field");
087            StoreWithArrayList();
088            objectContainer.Ext().Purge();
089            Close();
090            Open(Configure());
091            StartTimer();
092            query = objectContainer.Query();
093            query.Constrain(typeof(ItemWithArrayList));
094            query.Descend("_ids").Constrain(1).Contains();
095            result = query.Execute();
096            ItemWithArrayList itemWithArrayList = (ItemWithArrayList)result[0];
097            StopTimer("Querying object with ArrayList field: "
098                    + itemWithArrayList._ids.ToString());
099            Close();
100
101        }
