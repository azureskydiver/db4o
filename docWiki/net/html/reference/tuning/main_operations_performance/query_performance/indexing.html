Indexing

Using indexes is always a good idea to improve query performance. The following test illustrates index performance impact:

QueryPerformanceBenchmark.cs: RunIndexTest
01private void RunIndexTest()
02        ...{
03
04            Init();
05            System.Console.WriteLine("Storing " + _count + " objects with " + _depth
06                    + " levels of embedded objects:");
07
08            Clean();
09            System.Console.WriteLine(" - no index");
10            Open(Configure());
11            Store();
12            Close();
13            Open(Configure());
14            StartTimer();
15            IQuery query = objectContainer.Query();
16            query.Constrain(typeof(Item));
17            query.Descend("_name").Constrain("level1/2");
18            IList result = query.Execute();
19            Item item = (Item)result[0];
20            StopTimer("Querying object with string field: " + item._name);
21            Close();
22
23
24            System.Console.WriteLine(" - index on string field");
25            // Open to create index
26            Open(ConfigureIndex());
27            Close();
28            Open(Configure());
29            StartTimer();
30            query = objectContainer.Query();
31            query.Constrain(typeof(Item));
32            query.Descend("_name").Constrain("level1/2");
33            result = query.Execute();
34            item = (Item)result[0];
35            StopTimer("Querying object with string field: " + item._name);
36            Close();
37        }
