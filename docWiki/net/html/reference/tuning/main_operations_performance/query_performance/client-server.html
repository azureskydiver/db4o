Client-Server

In client-server use the connection speed can play an important role in the query performance.

QueryPerformanceBenchmark.cs: RunClientServerTest
01private void RunClientServerTest()
02        ...{
03
04            InitForClientServer();
05            Clean();
06            System.Console.WriteLine("Storing " + _count + " objects of depth " + _depth
07                    + " remotely:");
08            Open(ConfigureClientServer());
09            Store();
10            Close();
11            Open(ConfigureClientServer());
12            StartTimer();
13            IQuery query = objectContainer.Query();
14            query.Constrain(typeof(Item));
15            query.Descend("_name").Constrain("level1/1");
16            Item item = (Item)query.Execute().Next();
17            StopTimer("Select 1 object: " + item._name);
18            Close();
19
20            Init();
21            Clean();
22            System.Console.WriteLine("Storing " + _count + " objects of depth " + _depth
23                    + " locally:");
24            Open(ConfigureClientServer());
25            Store();
26            Close();
27            Open(ConfigureClientServer());
28            StartTimer();
29            query = objectContainer.Query();
30            query.Constrain(typeof(Item));
31            query.Descend("_name").Constrain("level1/1");
32            item = (Item)query.Execute().Next();
33            StopTimer("Select 1 object: " + item._name);
34            Close();
35        }
QueryPerformanceBenchmark.cs: InitForClientServer
1private void InitForClientServer()
2        ...{
3            _filePath = "performance.db4o";
4            _isClientServer = true;
5            _host = "localhost";
6        }
QueryPerformanceBenchmark.cs: ConfigureClientServer
1private IConfiguration ConfigureClientServer()
2        ...{
3            IConfiguration config = Db4oFactory.NewConfiguration();
4            config.Queries().EvaluationMode(QueryEvaluationMode.Immediate);
5            config.ClientServer().SingleThreadedClient(true);
6            return config;
7        }

Results from the test machine:

Storing 30000 objects of depth 3 remotely:

Store 90000 objects: 10725ms

Select 1 object: level1/1: 1763ms

Storing 10000 objects of depth 3 locally:

Store 30000 objects: 2904ms

Select 1 object: level1/1: 630ms

In order to improve the performance use Lazy or Snapshot evaluation modes.
