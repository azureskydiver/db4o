Inherited Objects

You can use a class base to query for inherited objects. This makes a query path a bit more complex and may result in a small performance degrade.

QueryPerformanceBenchmark.cs: RunInheritanceTest
01private void RunInheritanceTest()
02        ...{
03            Init();
04            Clean();
05            System.Console.WriteLine("Storing " + _count + " objects of depth " + _depth);
06            Open(Configure());
07            Store();
08            Close();
09            Open(Configure());
10            StartTimer();
11            IQuery query = objectContainer.Query();
12            query.Constrain(typeof(Item));
13            query.Descend("_name").Constrain("level1/1");
14            Item item = (Item)query.Execute().Next();
15            StopTimer("Select 1 object: " + item._name);
16            Close();
17
18            Clean();
19            System.Console.WriteLine("Storing " + _count + " inherited objects of depth "
20                    + _depth);
21            Open(Configure());
22            StoreInherited();
23            Close();
24            Open(Configure());
25            StartTimer();
26            // Query for item, inheriting objects should be included in the result
27            query = objectContainer.Query();
28            query.Constrain(typeof(Item));
29            query.Descend("_name").Constrain("level1/1");
30            item = (Item)query.Execute().Next();
31            StopTimer("Select 1 object: " + item._name);
32            Close();
33        }
