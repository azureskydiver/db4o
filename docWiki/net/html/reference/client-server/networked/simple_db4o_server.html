Simple db4o Server

Let's implement a simple standalone db4o server with a special client that can tell the server to shut itself down gracefully on demand.

First, both the client and the server need some shared configuration information. We will provide this using an interface:

[code=../../clientserverVB.zip#ServerConfiguration.vb]

35        ...{

37        }

40        /**//// <summary>
41        /// opens the IObjectServer, and waits forever until Close() is called
42        /// or a StopServer message is being received.
43        /// </summary>
44        public void RunServer()
45        ...{
46            lock(this)
47            ...{
48                // Using the messaging functionality to redirect all
49                // messages to this.processMessage
50                IServerConfiguration configuration = Db4oClientServer.NewServerConfiguration();
51                configuration.Networking.MessageRecipient = this;
52
53                IObjectServer db4oServer = Db4oClientServer.OpenServer(configuration, FileName, Port);
54                db4oServer.GrantAccess(User, Password);
55                
56                try
57                ...{
58                    if (! stop)
59                    ...{
60                        // wait forever until Close will change stop variable
61                        Monitor.Wait(this);   
62                    }
63                }
64                catch (Exception e)
65                ...{
66                    Console.WriteLine(e.ToString());
67                }
68                db4oServer.Close();
69            }
70        }
71    // end RunServer
72
73        /**//// <summary>
74        /// messaging callback
75        /// see com.db4o.messaging.MessageRecipient#ProcessMessage()
76        /// </summary>
77        public void ProcessMessage(IMessageContext context, object message)
78        ...{
79            if (message is StopServer)
80            ...{
81                Close();
82            }
83        }
84    // end ProcessMessage
85
86        /**//// <summary>
87        /// closes this server.
88        /// </summary>
89        public void Close()
90        ...{
91            lock(this)
92            ...{

And last but not least, the client that stops the server.

StopServer.cs
01/**//* Copyright (C) 2009 Versant Inc. http://www.db4o.com */
02using System;
03using Db4objects.Db4o;
04using Db4objects.Db4o.CS;
05using Db4objects.Db4o.Messaging;
06
07namespace Db4odoc.ClientServer
08...{
09    /**//// <summary>
10    /// stops the db4o Server started with StartServer.
11    /// This is done by opening a client connection
12    /// to the server and by sending a StopServer object as
13    /// a message. StartServer will react in it's
14    /// a message. StartServer will react in it's
15    /// processMessage method.
16    /// </summary>
17    public class StopServer : ServerSetup
18    ...{
19        /**//// <summary>
20        /// stops a db4o Server started with StartServer.
21        /// </summary>
22        /// <exception cref="Exception" />
23        public static void Main(string[] args)
24        ...{
25            IObjectContainer objectContainer = null;
26            try
27            ...{
28                // connect to the server
29                objectContainer = Db4oClientServer.OpenClient(Db4oClientServer.NewClientConfiguration(), Host, Port, User, Password);
30            }
31            catch (Exception e)
32            ...{
33                Console.WriteLine(e.ToString());
34            }
35
36            if (objectContainer != null)
37            ...{
38                // get the messageSender for the IObjectContainer 
39                IMessageSender messageSender = objectContainer.Ext()
40                    .Configure().ClientServer().GetMessageSender();
41
42                // send an instance of a StopServer object
43                messageSender.Send(new StopServer());
44
45                // close the IObjectContainer 
46                objectContainer.Close();
47            }
48        }
49        // end Main
50    }
51}
