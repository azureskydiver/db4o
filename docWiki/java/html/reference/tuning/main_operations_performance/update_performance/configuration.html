UpdatePerformanceBenchmark.java: configure
1private Configuration configure()...{
2      Configuration config = Db4o.newConfiguration();
3      // using MemoryIoAdapter improves the performance 
4      // by replacing the costly disk IO operations with 
5      // memory access
6        config.io(new MemoryIoAdapter());
7        return config;
8    }
UpdatePerformanceBenchmark.java: configureCascade
1private Configuration configureCascade()...{
2      Configuration config = Db4o.newConfiguration(); 
3      // CascadeOnUpdate can be a performance-killer for 
4      // deep object hierarchies
5        config.objectClass(Item.class).cascadeOnUpdate(true);
6        return config;
7    }
UpdatePerformanceBenchmark.java: configureTP
1private Configuration configureTP()...{
2      Configuration config = Db4o.newConfiguration();
3      // With Transparent Persistence enabled only modified
4      // objects are written to disk. This allows to achieve 
5      // better performance
6        config.objectClass(Item.class).cascadeOnUpdate(true);
7      return config;
8    }
