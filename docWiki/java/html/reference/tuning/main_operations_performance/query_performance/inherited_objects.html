Inherited Objects

You can use a class base to query for inherited objects. This makes a query path a bit more complex and may result in a small performance degrade.

QueryPerformanceBenchmark.java: runInheritanceTest
01private void runInheritanceTest() ...{
02    init();
03    clean();
04    System.out.println("Storing " + _count + " objects of depth " + _depth);
05    open(configure());
06    store();
07    close();
08    open(configure());
09    startTimer();
10    Query query = objectContainer.query();
11    query.constrain(Item.class);
12    query.descend("_name").constrain("level1/1");
13    Item item = (Item) query.execute().next();
14    stopTimer("Select 1 object: " + item._name);
15    close();
16
17    clean();
18    System.out.println("Storing " + _count + " inherited objects of depth "
19        + _depth);
20    open(configure());
21    storeInherited();
22    close();
23    open(configure());
24    startTimer();
25    // Query for item, inheriting objects should be included in the result
26    query = objectContainer.query();
27    query.constrain(Item.class);
28    query.descend("_name").constrain("level1/1");
29    item = (Item) query.execute().next();
30    stopTimer("Select 1 object: " + item._name);
31    close();
32  }
