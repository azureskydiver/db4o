10  <class name="f1.collection.set.Pilot">
11    <id column="typed_id" type="long">
12      <generator class="native"/>
13    </id>
14
15    <property name="name"/>
16
17    <set name="cars" table="cars">
18      <key column="pilotId"/>
19      <one-to-many class="f1.collection.Car"/>
20    </set>
21  </class>
22</hibernate-mapping>
