Creating your own reflector

By default db4o uses JdkReflector(Java) or NetReflector (.NET) as a GenericReflector delegate.

However, the programmer can instruct db4o to use a specially designed reflection implementation:

Java: Db4o.configure().reflectWith(reflector)

where reflector is one of the available reflectors or your own reflector implementation.

At present db4o comes with SelfReflector, which was designed for environments, which do not have built-in support for reflections (J2ME for example). In this implementation all the classes' information is stored in special registry. User classes should implement self_get and self_set methods to be registered individually and become "known" to SelfReflector.

Specific reflectors can be written for special usecases.

Let's look how to create a reflector. Remember that db4o relies on reflector to read the database, so errors in reflector may prevent your database from opening.

To keep things simple we will write a LoggingReflector, its only difference from standard reflector is that information about loaded classes is outputted to console. All reflectors used by db4o should implement com.db4o.reflect.Reflector/Db4objects.Db4o.Reflect.IReflector interface.

LoggingReflector.java
001/**//* Copyright (C) 2004   db4objects Inc.   http://www.db4o.com */
002
003package com.db4odoc.reflections;
004
005import com.db4o.internal.Platform4;
006import com.db4o.reflect.ReflectArray;
007import com.db4o.reflect.ReflectClass;
008import com.db4o.reflect.Reflector;
009import com.db4o.reflect.ReflectorConfiguration;
010import com.db4o.reflect.jdk.ClassLoaderJdkLoader;
011import com.db4o.reflect.jdk.JavaReflectClass;
012import com.db4o.reflect.jdk.JdkClass;
013import com.db4o.reflect.jdk.JdkLoader;
014import com.db4o.reflect.jdk.JdkReflector;
015
016/** *//**
017 * db4o wrapper for JDK reflector functionality
018 * 
019 * @see com.db4o.ext.ExtObjectContainer#reflector()
020 * @see com.db4o.reflect.generic.GenericReflector
021 * 
022 * @sharpen.ignore
023 */
024public class LoggingReflector implements Reflector ...{
025
026  private final JdkLoader _classLoader;
027  protected Reflector _parent;
028  private ReflectArray _array;
029  private ReflectorConfiguration _config;
030
031  /** *//**
032   * Constructor
033   * 
034   * @param classLoader
035   *            class loader
036   */
037  public LoggingReflector(ClassLoader classLoader) ...{
038    this(new ClassLoaderJdkLoader(classLoader));
039  }
040
041  /** *//**
042   * Constructor
043   * 
044   * @param classLoader
045   *            class loader
046   */
047  public LoggingReflector(JdkLoader classLoader) ...{
048    this(classLoader, null);
049  }
050
051  private LoggingReflector(JdkLoader classLoader,
052      ReflectorConfiguration config) ...{
053    _classLoader = classLoader;
054    _config = config;
055  }
056
057  /** *//**
058   * ReflectArray factory
059   * 
060   * @return ReflectArray instance
061   */
062  public ReflectArray array() ...{
063    if (_array == null) ...{
064      _array = new LoggingArray(parent());
065    }
066    return _array;
067  }
068
069  /** *//**
070   * Creates a copy of the object
071   * 
072   * @param obj
073   *            object to copy
074   * @return object copy
075   */
076  public Object deepClone(Object obj) ...{
077    return new LoggingReflector(_classLoader, _config);
078  }
079
080  /** *//**
081   * Returns ReflectClass for the specified class
082   * 
083   * @param clazz
084   *            class
085   * @return ReflectClass for the specified class
086   */
087  public ReflectClass forClass(Class clazz) ...{
088    ReflectClass rc = createClass(clazz);
089    System.out.println("forClass: " + clazz + " -> "
090        + (rc == null ? "" : rc.getName()));
091
092    return rc;
093  }
094
095  /** *//**
096   * Returns ReflectClass for the specified class name
097   * 
098   * @param className
099   *            class name
100   * @return ReflectClass for the specified class name
101   */
102  public ReflectClass forName(String className) ...{
103    Class clazz = _classLoader.loadClass(className);
104    ReflectClass rc = createClass(clazz);
105    System.out.println("forName: " + className + " -> "
106        + (rc == null ? "" : rc.getName()));
107    return rc;
108
109  }
110
111  /** *//**
112   * creates a Class reflector when passed a class. This method is protected
113   * to allow overriding in cusom reflectors that override JdkReflector.
114   * 
115   * @param clazz
116   *            the class
117   * @return the class reflector
118   */
119  protected JdkClass createClass(Class clazz) ...{
120    if (clazz == null) ...{
121      return null;
122    }
123    JdkReflector jdkReflector = new JdkReflector(this.getClass()
124        .getClassLoader());
125    jdkReflector.configuration(_config);
126
127    JdkClass rc = new JdkClass(parent(), jdkReflector, clazz);
128    return rc;
129  }
130
131  /** *//**
132   * Returns ReflectClass for the specified class object
133   * 
134   * @param a_object
135   *            class object
136   * @return ReflectClass for the specified class object
137   */
138  public ReflectClass forObject(Object a_object) ...{
139    if (a_object == null) ...{
140      return null;
141    }
142    ReflectClass rc = parent().forClass(a_object.getClass());
143    System.out.println("forObject:" + a_object + " -> "
144        + (rc == null ? "" : rc.getName()));
145    return rc;
146  }
147
148  /** *//**
149   * Method stub. Returns false.
150   */
151  public boolean isCollection(ReflectClass candidate) ...{
152    return false;
153  }
154
155  /** *//**
156   * Method stub. Returns false.
157   */
158  public boolean methodCallsSupported() ...{
159    return true;
160  }
161
162  /** *//**
163   * Sets parent reflector
164   * 
165   * @param reflector
166   *            parent reflector
167   */
168  public void setParent(Reflector reflector) ...{
169    _parent = reflector;
170  }
171
172  /** *//**
173   * Creates ReflectClass[] array from the Class[] array using the reflector
174   * specified
175   * 
176   * @param reflector
177   *            reflector to use
178   * @param clazz
179   *            class
180   * @return ReflectClass[] array
181   */
182  public static ReflectClass[] toMeta(Reflector reflector, Class[] clazz) ...{
183    ReflectClass[] claxx = null;
184    if (clazz != null) ...{
185      claxx = new ReflectClass[clazz.length];
186      for (int i = 0; i < clazz.length; i++) ...{
187        if (clazz[i] != null) ...{
188          claxx[i] = reflector.forClass(clazz[i]);
189        }
190      }
191    }
192    return claxx;
193  }
194
195  /** *//**
196   * Creates Class[] array from the ReflectClass[] array
197   * 
198   * @param claxx
199   *            ReflectClass array
200   * @return Class[] array
201   */
202  static Class[] toNative(ReflectClass[] claxx) ...{
203    Class[] clazz = null;
204    if (claxx != null) ...{
205      clazz = new Class[claxx.length];
206      for (int i = 0; i < claxx.length; i++) ...{
207        clazz[i] = toNative(claxx[i]);
208      }
209    }
210    return clazz;
211  }
212
213  /** *//**
214   * Translates a ReflectClass into a native Class
215   * 
216   * @param claxx
217   *            ReflectClass to translate
218   * @return Class
219   */
220  public static Class toNative(ReflectClass claxx) ...{
221    if (claxx == null) ...{
222      return null;
223    }
224    System.out.println("toNative: " + claxx.getName());
225    if (claxx instanceof JavaReflectClass) ...{
226      return ((JavaReflectClass) claxx).getJavaClass();
227    }
228    ReflectClass d = claxx.getDelegate();
229    if (d == claxx) ...{
230      return null;
231    }
232    return toNative(d);
233  }
234
235  public void configuration(ReflectorConfiguration config) ...{
236    _config = config;
237  }
238
239  public ReflectorConfiguration configuration() ...{
240    return _config;
241  }
242
243  Object nullValue(ReflectClass clazz) ...{
244    return Platform4.nullValue(toNative(clazz));
245  }
246
247  private Reflector parent() ...{
248    if (_parent == null) ...{
249      return this;
250    }
251    return _parent;
252  }
253
254}

It is easy to see that this reflector provides the same functionality as JdkReflector or NetReflector extended by console output. The following simple test will show how it works:

ReflectorExample.java: testReflector
01private static void testReflector()
02  ...{
03    LoggingReflector logger = new LoggingReflector(Db4o.class.getClassLoader());
04    Configuration configuration = Db4o.newConfiguration();
05    configuration.reflectWith(logger);
06    ObjectContainer container = Db4o.openFile(configuration, DB4O_FILE_NAME);
07    try ...{
08      ReflectClass rc  = container.ext().reflector().forName(Car.class.getName());
09      System.out.println("Reflected class: " + rc);
10    } finally ...{
11      container.close();
12    }
13  }

The output can help you to track all the loaded classes.

Reflection is a powerful tool, which plays a fundamental role in db4o. Understanding reflection will help you to understand the whole db4o functionality in detail.
